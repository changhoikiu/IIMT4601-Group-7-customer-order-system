POST Customer_Enquiry (enquiryType: Order a Book)
{
    Customer_Name : form.name;
    Customer_Email: form.email;
    Customer_Phone_No: form.phoneNo;
    Enquiry_timestamp: TIMESTAMP;
    Enquiry_Subject: enquiryType.name;
    Enquiry_Message: form.title + form.authors + form.genre; --String
}

POST Customer_Enquiry (enquiryType: Cancel an Order)
{
    Customer_Name : form.name;
    Customer_Email: form.email;
    Customer_Phone_No: form.phoneNo;
    Enquiry_timestamp: TIMESTAMP;
    Enquiry_Subject: enquiryType.name;
    Enquiry_Message: form.title + form.authors; --String
    Reservation_ID: form.reservationID
}

POST Customer_Enquiry (enquiryType: Other Enquiries)
{
    Customer_Name : form.name;
    Customer_Email: form.email;
    Customer_Phone_No: form.phoneNo;
    Enquiry_timestamp: TIMESTAMP;
    Enquiry_Subject: enquiryType.name;
    Enquiry_Message: form.message;
    Reservation_ID: null;
}
